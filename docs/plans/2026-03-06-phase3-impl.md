// ImpVMRunnerPoolSpec defines a pool of ephemeral CI runner VMs.
type ImpVMRunnerPoolSpec struct {
	// TemplateName references an ImpVMTemplate in the same namespace.
	TemplateName string `json:"templateName"`

	// Platform configures the CI platform integration.
	Platform RunnerPlatformSpec `json:"platform"`

	// RunnerLayer is the OCI image containing the runner binary.
	// When set, it is composited on top of the template image at boot time.
	// Omit if the runner binary is already baked into the template image.
	// +optional
	RunnerLayer string `json:"runnerLayer,omitempty"`

	// Labels are applied to runner registrations on the CI platform.
	// +optional
	Labels []string `json:"labels,omitempty"`

	// Scaling controls how many runner VMs are maintained.
	Scaling RunnerScalingSpec `json:"scaling,omitempty"`

	// JobDetection configures how the operator discovers queued jobs.
	JobDetection RunnerJobDetectionSpec `json:"jobDetection,omitempty"`
}

// RunnerPlatformSpec identifies the CI platform and credentials.

	// Type selects the CI platform driver.
	// +kubebuilder:validation:Enum=github-actions;gitlab;forgejo
	Type string `json:"type"`

	// +optional
	ServerURL string `json:"serverURL,omitempty"`

package runner

import "context"

// JITConfig is a one-time runner registration token + endpoint issued by the platform.
type JITConfig struct {
	// EncodedConfig is the base64-encoded JIT config string passed to the runner binary.
	EncodedConfig string
	// RunnerName is the suggested runner name (platform-generated).
	RunnerName string
}

// PlatformDriver knows how to interact with a specific CI platform.
// Each method must be idempotent and safe to call concurrently.
type PlatformDriver interface {
	// GetJITConfig exchanges the stored credential for a one-time runner config.
	// The returned JITConfig is valid for a single runner registration.
	GetJITConfig(ctx context.Context) (*JITConfig, error)

	// QueueDepth returns the number of jobs currently queued and waiting for a runner.
	// Returns 0 when no jobs are waiting or when the platform API is unreachable.
	QueueDepth(ctx context.Context) (int, error)

	// ValidateWebhook checks the HMAC signature of an inbound webhook payload.
	// Returns an error if the signature is invalid or the payload cannot be parsed.
	// Returns the number of queued jobs mentioned in the payload (0 if not a job event).
	ValidateWebhook(payload []byte, signature string) (int, error)
}

package runner

import (
	"context"
	"crypto/hmac"
	"crypto/sha256"
	"encoding/hex"
	"encoding/json"
	"errors"
	"fmt"
	"strings"

	"github.com/google/go-github/v67/github"
	"golang.org/x/oauth2"
)

// GitHubDriver implements PlatformDriver for GitHub Actions.
// It supports org-level and repo-level JIT runner registration.
type GitHubDriver struct {
	client    *github.Client
	org       string // empty when repo-level
	owner     string // repo owner (only used when repo is set)
	repo      string // repo name (only used when org is empty)
	hmacSecret []byte
}

// NewGitHubDriver creates a driver for github.com.
// token must be a PAT with actions:write scope or a fine-grained token with runners write.
// scope must be "org:<org>" or "repo:<owner>/<repo>".
func NewGitHubDriver(token, scope string, hmacSecret []byte) (*GitHubDriver, error) {
	ts := oauth2.StaticTokenSource(&oauth2.Token{AccessToken: token})
	httpClient := oauth2.NewClient(context.Background(), ts)
	client := github.NewClient(httpClient)
	return newGitHubDriverWithClient(client, scope, hmacSecret)
}

// NewForgejoDriver creates a driver for a Forgejo instance.
// Forgejo implements the GitHub Actions runner API at a custom serverURL.
func NewForgejoDriver(token, serverURL, scope string, hmacSecret []byte) (*GitHubDriver, error) {
	ts := oauth2.StaticTokenSource(&oauth2.Token{AccessToken: token})
	httpClient := oauth2.NewClient(context.Background(), ts)
	client, err := github.NewClient(httpClient).WithAuthToken(token)
	if err != nil {
		return nil, err
	}
	baseURL := strings.TrimRight(serverURL, "/") + "/api/v1/"
	uploadURL := baseURL
	client, err = client.WithEnterpriseURLs(baseURL, uploadURL)
	if err != nil {
		return nil, fmt.Errorf("forgejo client: %w", err)
	}
	return newGitHubDriverWithClient(client, scope, hmacSecret)
}

func newGitHubDriverWithClient(client *github.Client, scope string, hmacSecret []byte) (*GitHubDriver, error) {
	d := &GitHubDriver{client: client, hmacSecret: hmacSecret}
	switch {
	case strings.HasPrefix(scope, "org:"):
		d.org = strings.TrimPrefix(scope, "org:")
	case strings.HasPrefix(scope, "repo:"):
		parts := strings.SplitN(strings.TrimPrefix(scope, "repo:"), "/", 2)
		if len(parts) != 2 {
			return nil, fmt.Errorf("invalid repo scope %q: expected owner/repo", scope)
		}
		d.owner, d.repo = parts[0], parts[1]
	default:
		return nil, fmt.Errorf("invalid scope %q: must start with org: or repo:", scope)
	}
	return d, nil
}

func (d *GitHubDriver) GetJITConfig(ctx context.Context) (*JITConfig, error) {
	req := &github.GenerateJITConfigRequest{
		Name:          "imp-runner",
		RunnerGroupID: 1, // default group
		Labels:        []string{"self-hosted"},
	}
	var resp *github.JITRunnerConfig
	var err error
	if d.org != "" {
		resp, _, err = d.client.Actions.GenerateOrgJITConfig(ctx, d.org, req)
	} else {
		resp, _, err = d.client.Actions.GenerateRepoJITConfig(ctx, d.owner, d.repo, req)
	}
	if err != nil {
		return nil, fmt.Errorf("GetJITConfig: %w", err)
	}
	return &JITConfig{
		EncodedConfig: resp.GetEncodedJITConfig(),
		RunnerName:    resp.Runner.GetName(),
	}, nil
}

func (d *GitHubDriver) QueueDepth(ctx context.Context) (int, error) {
	var runs *github.WorkflowRuns
	var err error
	opts := &github.ListWorkflowRunsOptions{Status: "queued"}
	if d.org != "" {
		runs, _, err = d.client.Actions.ListOrganizationWorkflowRuns(ctx, d.org, opts)
	} else {
		runs, _, err = d.client.Actions.ListRepositoryWorkflowRuns(ctx, d.owner, d.repo, opts)
	}
	if err != nil {
		return 0, nil // best-effort; don't fail the reconcile
	}
	return runs.GetTotalCount(), nil
}

func (d *GitHubDriver) ValidateWebhook(payload []byte, signature string) (int, error) {
	if !d.verifyHMAC(payload, signature) {
		return 0, errors.New("invalid webhook signature")
	}
	// Parse the workflow_job event.
	var event struct {
		Action      string `json:"action"`
		WorkflowJob struct {
			Status string `json:"status"`
		} `json:"workflow_job"`
	}
	if err := json.Unmarshal(payload, &event); err != nil {
		return 0, nil // not a job event; ignore
	}
	if event.Action == "queued" {
		return 1, nil
	}
	return 0, nil
}

func (d *GitHubDriver) verifyHMAC(payload []byte, signature string) bool {
	if len(d.hmacSecret) == 0 {
		return true // no secret configured → accept all (testing only)
	}
	mac := hmac.New(sha256.New, d.hmacSecret)
	mac.Write(payload)
	expected := "sha256=" + hex.EncodeToString(mac.Sum(nil))
	return hmac.Equal([]byte(expected), []byte(signature))
}

import (

	"strings"

	gitlab "github.com/xanzy/go-gitlab"
)

// GitLabDriver implements PlatformDriver for GitLab CI.
type GitLabDriver struct {
	client     *gitlab.Client
	groupPath  string // group-level runner
	projectID  string // project-level runner (empty when group-level)
	hmacSecret []byte
}

// NewGitLabDriver creates a driver for a GitLab instance.
// token is a PAT with create_runner scope.
// scope must be "group:<path>" or "project:<id>".
// serverURL is the GitLab base URL, e.g. "https://gitlab.com".
func NewGitLabDriver(token, serverURL, scope string, hmacSecret []byte) (*GitLabDriver, error) {
	opts := []gitlab.ClientOptionFunc{}
	if serverURL != "" {
		opts = append(opts, gitlab.WithBaseURL(serverURL))
	}
	client, err := gitlab.NewClient(token, opts...)
	if err != nil {
		return nil, fmt.Errorf("gitlab client: %w", err)
	}
	d := &GitLabDriver{client: client, hmacSecret: hmacSecret}
	switch {
	case strings.HasPrefix(scope, "group:"):
		d.groupPath = strings.TrimPrefix(scope, "group:")
	case strings.HasPrefix(scope, "project:"):
		d.projectID = strings.TrimPrefix(scope, "project:")
	default:
		return nil, fmt.Errorf("invalid scope %q: must start with group: or project:", scope)
	}
	return d, nil
}

func (d *GitLabDriver) GetJITConfig(ctx context.Context) (*JITConfig, error) {
	// GitLab uses runner authentication tokens, not JIT configs.
	// We create a new runner registration and return its token as EncodedConfig.
	runnerType := "group_type"
	scopeID := 0
	if d.projectID != "" {
		runnerType = "project_type"
	}
	opts := &gitlab.CreateRunnerOptions{
		RunnerType: gitlab.Ptr(runnerType),
		GroupID:    gitlab.Ptr(scopeID),
		Description: gitlab.Ptr("imp-runner"),
		TagList:    &[]string{"self-hosted"},
	}
	runner, _, err := d.client.Runners.CreateRunner(opts, gitlab.WithContext(ctx))
	if err != nil {
		return nil, fmt.Errorf("GitLab CreateRunner: %w", err)
	}
	return &JITConfig{
		EncodedConfig: runner.Token,
		RunnerName:    fmt.Sprintf("imp-runner-%d", runner.ID),
	}, nil
}

func (d *GitLabDriver) QueueDepth(ctx context.Context) (int, error) {
	// GitLab doesn't expose a simple queue depth endpoint without admin scope.
	// Return 0 (rely on webhook for job detection).
	return 0, nil
}

func (d *GitLabDriver) ValidateWebhook(payload []byte, signature string) (int, error) {
	if len(d.hmacSecret) > 0 {
		mac := hmac.New(sha256.New, d.hmacSecret)
		mac.Write(payload)
		expected := hex.EncodeToString(mac.Sum(nil))
		if !hmac.Equal([]byte(expected), []byte(signature)) {
			return 0, errors.New("invalid GitLab webhook signature")
		}
	}
	var event struct {
		ObjectKind string `json:"object_kind"`
		BuildStatus string `json:"build_status"`
	}
	if err := json.Unmarshal(payload, &event); err != nil {
		return 0, nil
	}
	if event.ObjectKind == "build" && event.BuildStatus == "pending" {
		return 1, nil
	}
	return 0, nil
}

type ImpVMRunnerPoolReconciler struct {
	client.Client
	Scheme *runtime.Scheme

// +kubebuilder:rbac:groups=imp.dev,resources=impvmrunnerpools,verbs=get;list;watch;create;update;patch;delete
// +kubebuilder:rbac:groups=imp.dev,resources=impvmrunnerpools/status,verbs=get;update;patch
// +kubebuilder:rbac:groups=imp.dev,resources=impvmtemplates,verbs=get;list;watch
// +kubebuilder:rbac:groups=imp.dev,resources=impvms,verbs=get;list;watch;create;delete

func (r *ImpVMRunnerPoolReconciler) Reconcile(ctx context.Context, req ctrl.Request) (ctrl.Result, error) {
	log := logf.FromContext(ctx)

	pool := &impv1alpha1.ImpVMRunnerPool{}
	if err := r.Get(ctx, req.NamespacedName, pool); err != nil {
		return ctrl.Result{}, client.IgnoreNotFound(err)
	}

	// Fetch template.
	tpl := &impv1alpha1.ImpVMTemplate{}
	if err := r.Get(ctx, client.ObjectKey{Namespace: pool.Namespace, Name: pool.Spec.TemplateName}, tpl); err != nil {
		if apierrors.IsNotFound(err) {
			log.Info("ImpVMTemplate not found, requeueing", "template", pool.Spec.TemplateName)
			return ctrl.Result{RequeueAfter: 30 * time.Second}, nil
		}
		return ctrl.Result{}, err
	}

	// List pool members.
	vmList := &impv1alpha1.ImpVMList{}
	if err := r.List(ctx, vmList,
		client.InNamespace(pool.Namespace),
		client.MatchingLabels{impv1alpha1.LabelRunnerPool: pool.Name},
	); err != nil {
		return ctrl.Result{}, err
	}

	// Delete terminal VMs — runners are single-use.
	for i := range vmList.Items {
		vm := &vmList.Items[i]
		switch vm.Status.Phase {
		case impv1alpha1.VMPhaseSucceeded, impv1alpha1.VMPhaseFailed, impv1alpha1.VMPhaseRetryExhausted:
			if err := r.Delete(ctx, vm); err != nil && !apierrors.IsNotFound(err) {
				return ctrl.Result{}, err
			}
			log.Info("deleted terminal runner VM", "vm", vm.Name, "phase", vm.Status.Phase)
		}
	}

	// Re-list after deletion.
	if err := r.List(ctx, vmList,
		client.InNamespace(pool.Namespace),
		client.MatchingLabels{impv1alpha1.LabelRunnerPool: pool.Name},
	); err != nil {
		return ctrl.Result{}, err
	}

	// Count active VMs.
	var activeCount int32
	for i := range vmList.Items {
		phase := vmList.Items[i].Status.Phase
		switch phase {
		case impv1alpha1.VMPhaseSucceeded, impv1alpha1.VMPhaseFailed, impv1alpha1.VMPhaseRetryExhausted:
			// skip — just deleted above
		default:
			activeCount++
		}
	}

	// Create VMs to reach minIdle, up to maxConcurrent.
	toCreate := pool.Spec.Scaling.MinIdle - activeCount
	if toCreate < 0 {
		toCreate = 0
	}
	available := pool.Spec.Scaling.MaxConcurrent - activeCount
	if toCreate > available {
		toCreate = available
	}

	for i := int32(0); i < toCreate; i++ {
		if err := r.createRunnerVM(ctx, pool, tpl); err != nil {
			return ctrl.Result{}, err
		}
	}
	if toCreate > 0 {
		log.Info("created runner VMs", "pool", pool.Name, "count", toCreate)
	}

	// Patch status.
	base := pool.DeepCopy()
	pool.Status.ActiveCount = activeCount
	pool.Status.IdleCount = activeCount // approximation until runner lifecycle tracking is added
	if err := r.Status().Patch(ctx, pool, client.MergeFrom(base)); err != nil {
		return ctrl.Result{}, err
	}

	requeueAfter := 30 * time.Second
	if pool.Spec.JobDetection.Polling.Enabled && pool.Spec.JobDetection.Polling.IntervalSeconds > 0 {
		requeueAfter = time.Duration(pool.Spec.JobDetection.Polling.IntervalSeconds) * time.Second
	}
	return ctrl.Result{RequeueAfter: requeueAfter}, nil
}

```go
// BuildComposite builds an ext4 rootfs by overlaying extraLayerRefs images on top of imageRef.
// Layers are applied in order: imageRef layers first, then each extraLayerRef in order.
// The cache key is the hash of all combined layer digests.
// When extraLayerRefs is empty, this is equivalent to Build.
func (b *Builder) BuildComposite(ctx context.Context, imageRef string, extraLayerRefs []string, opts ...BuildOption) (string, error) {
	if len(extraLayerRefs) == 0 {
		return b.Build(ctx, imageRef, opts...)
	}

	// Pull base image.
	base, err := b.pullImage(ctx, imageRef)
	if err != nil {
		return "", fmt.Errorf("pull base %q: %w", imageRef, err)
	}

	// Compute composite cache key from all digests.
	baseDigest, err := base.Digest()
	if err != nil {
		return "", fmt.Errorf("base digest: %w", err)
	}
	digestParts := []string{baseDigest.Hex}

	// Pull extra layer images and collect their layers.
	var extraLayers []v1.Layer
	for _, ref := range extraLayerRefs {
		img, err := b.pullImage(ctx, ref)
		if err != nil {
			return "", fmt.Errorf("pull extra layer %q: %w", ref, err)
		}
		d, err := img.Digest()
		if err != nil {
			return "", fmt.Errorf("extra layer digest: %w", err)
		}
		digestParts = append(digestParts, d.Hex)
		layers, err := img.Layers()
		if err != nil {
			return "", fmt.Errorf("extra layers: %w", err)
		}
		extraLayers = append(extraLayers, layers...)
	}

	// Build composite cache key.
	import_crypto_sha256 := sha256.New()
	for _, p := range digestParts {
		import_crypto_sha256.Write([]byte(p))
	}
	cacheKey := hex.EncodeToString(import_crypto_sha256.Sum(nil))
	if len(opts) > 0 {
		cacheKey += "-ga"
	}
	dest := b.cachePath(cacheKey)
	if _, err := os.Stat(dest); err == nil {
		return dest, nil
	}

	if err := b.ensureCacheDir(); err != nil {
		return "", fmt.Errorf("cache dir: %w", err)
	}

	// Extract base image layers.
	tmpDir, err := os.MkdirTemp("", "imp-rootfs-composite-*")
	if err != nil {
		return "", fmt.Errorf("temp dir: %w", err)
	}
	defer os.RemoveAll(tmpDir)

	if err := extractImage(ctx, base, tmpDir); err != nil {
		return "", fmt.Errorf("extract base: %w", err)
	}

	// Extract extra layers on top (in order).
	for i, layer := range extraLayers {
		if err := extractLayer(ctx, layer, tmpDir); err != nil {
			return "", fmt.Errorf("extract extra layer %d: %w", i, err)
		}
	}

	// Apply build options (e.g. guest agent injection).
	for _, opt := range opts {
		if err := opt(tmpDir); err != nil {
			return "", fmt.Errorf("build option: %w", err)
		}
	}

	// Build ext4.
	return dest, buildExt4(ctx, tmpDir, dest)
}
```

//go:build linux

package network

import (
	"context"
	"fmt"
	"net"

	"github.com/vishvananda/netlink"
	logf "sigs.k8s.io/controller-runtime/pkg/log"
)

// EnsureVXLAN creates a VXLAN interface named vxlanIface with VNI vni,
// bound to the host IP localIP. Idempotent — does nothing if already present.
func (m *LinuxNetManager) EnsureVXLAN(ctx context.Context, vxlanIface string, vni int, localIP string) error {
	log := logf.FromContext(ctx)

	// Check if the link already exists.
	if _, err := netlink.LinkByName(vxlanIface); err == nil {
		return nil // already present
	}

	srcIP := net.ParseIP(localIP)
	if srcIP == nil {
		return fmt.Errorf("invalid localIP %q", localIP)
	}

	vxlan := &netlink.Vxlan{
		LinkAttrs: netlink.LinkAttrs{Name: vxlanIface},
		VxlanId:   vni,
		SrcAddr:   srcIP,
		Port:      4789,
		Learning:  false, // static FDB only
	}
	if err := netlink.LinkAdd(vxlan); err != nil {
		return fmt.Errorf("add VXLAN %q: %w", vxlanIface, err)
	}
	link, err := netlink.LinkByName(vxlanIface)
	if err != nil {
		return fmt.Errorf("get VXLAN link %q: %w", vxlanIface, err)
	}
	if err := netlink.LinkSetUp(link); err != nil {
		return fmt.Errorf("set VXLAN %q up: %w", vxlanIface, err)
	}
	log.Info("created VXLAN interface", "iface", vxlanIface, "vni", vni)
	return nil
}

// SyncFDB replaces the FDB entries on vxlanIface with entries.
// Stale entries (present locally but not in entries) are removed.
func (m *LinuxNetManager) SyncFDB(ctx context.Context, vxlanIface string, entries []FDBEntry) error {
	log := logf.FromContext(ctx)

	link, err := netlink.LinkByName(vxlanIface)
	if err != nil {
		return fmt.Errorf("get VXLAN link %q: %w", vxlanIface, err)
	}

	// Build desired set.
	desired := make(map[string]string, len(entries)) // MAC → NodeIP
	for _, e := range entries {
		desired[e.MAC] = e.NodeIP
	}

	// Get existing FDB entries.
	existing, err := netlink.NeighList(link.Attrs().Index, netlink.FAMILY_ALL)
	if err != nil {
		return fmt.Errorf("list FDB %q: %w", vxlanIface, err)
	}

	// Remove stale entries.
	for _, n := range existing {
		mac := n.HardwareAddr.String()
		if _, ok := desired[mac]; !ok {
			if err := netlink.NeighDel(&n); err != nil {
				log.V(1).Info("FDB delete failed (ignored)", "mac", mac, "err", err)
			}
		}
	}

	// Add missing entries.
	existingSet := make(map[string]struct{}, len(existing))
	for _, n := range existing {
		existingSet[n.HardwareAddr.String()] = struct{}{}
	}
	for mac, nodeIP := range desired {
		if _, ok := existingSet[mac]; ok {
			continue // already present
		}
		hw, err := net.ParseMAC(mac)
		if err != nil {
			log.V(1).Info("invalid MAC in VTEP table (skipping)", "mac", mac)
			continue
		}
		neigh := &netlink.Neigh{
			LinkIndex:    link.Attrs().Index,
			Family:       netlink.FAMILY_ALL,
			State:        netlink.NUD_PERMANENT,
			Type:         netlink.NDA_DST,
			HardwareAddr: hw,
			IP:           net.ParseIP(nodeIP),
		}
		if err := netlink.NeighAdd(neigh); err != nil {
			log.V(1).Info("FDB add failed (ignored)", "mac", mac, "nodeIP", nodeIP, "err", err)
		}
	}
	return nil
}
```

// After VM reaches Running — register VTEP entry.
func (r *ImpVMReconciler) registerVTEP(ctx context.Context, vm *impv1alpha1.ImpVM, netInfo *network.NetworkInfo) error {
	if vm.Spec.NetworkRef == nil || netInfo == nil {
		return nil
	}
	net := &impv1alpha1.ImpNetwork{}
	if err := r.Client.Get(ctx, client.ObjectKey{
		Namespace: vm.Namespace,
		Name:      vm.Spec.NetworkRef.Name,
	}, net); err != nil {
		return client.IgnoreNotFound(err)
	}

	entry := impv1alpha1.VTEPEntry{
		NodeIP: r.NodeIP,
		VMIP:   netInfo.IP,
		VMMAC:  netInfo.MACAddr,
		VMUID:  string(vm.UID),
	}

	// Add entry if not already present.
	for _, e := range net.Status.VTEPTable {
		if e.VMUID == entry.VMUID {
			return nil // already registered
		}
	}

	base := net.DeepCopy()
	net.Status.VTEPTable = append(net.Status.VTEPTable, entry)
	return r.Client.Status().Patch(ctx, net, client.MergeFrom(base))
}
```

	"k8s.io/apimachinery/pkg/runtime/schema"

var ciliumExternalWorkloadGVR = schema.GroupVersionResource{
	Group:    "cilium.io",
	Version:  "v2",
	Resource: "ciliumexternalworkloads",
}

var ciliumExternalWorkloadGVK = schema.GroupVersionKind{
	Group:   "cilium.io",
	Version: "v2",
	Kind:    "CiliumExternalWorkload",
}

// newCiliumExternalWorkload constructs a CiliumExternalWorkload for the given VM.
// ipv4AllocCIDR is a /32 CIDR for the VM's assigned IP.
func newCiliumExternalWorkload(vmNamespace, vmName, ipv4AllocCIDR string) *unstructured.Unstructured {
	obj := &unstructured.Unstructured{}
	obj.SetGroupVersionKind(ciliumExternalWorkloadGVK)
	obj.SetName(vmNamespace + "--" + vmName) // cluster-scoped; namespace in name
	_ = unstructured.SetNestedField(obj.Object, ipv4AllocCIDR, "spec", "ipv4AllocCIDR")
	return obj
}

// ciliumPresent checks whether the CiliumExternalWorkload CRD is registered.
func ciliumPresent(mapper meta.RESTMapper) bool {
	_, err := mapper.RESTMapping(ciliumExternalWorkloadGVK.GroupKind(), ciliumExternalWorkloadGVK.Version)
	return err == nil
}

Imp works with any CNI plugin for basic networking within a single node. For **cross-node VM networking** and **NetworkPolicy enforcement**, Imp provides built-in support for [Cilium](https://cilium.io) only.
